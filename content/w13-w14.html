class Seat:
    def __init__(self, row, col, student_id):
        self.row = row
        self.col = col
        self.student_id = student_id

def parse_seat_info(seat_info):
    seat_info = seat_info.split(',')
    row = int(seat_info[0][1])
    col = int(seat_info[1][0])
    student_id = seat_info[2].strip()
    return Seat(row, col, student_id)

def print_seating_arrangement(seating_arrangement):
    for row in seating_arrangement:
        row_str = ''
        for seat in row:
            if seat:
                row_str += f"{seat.row},{seat.col} {seat.student_id} | "
            else:
                row_str += "空位 | "
        print(row_str.rstrip(" | "))
